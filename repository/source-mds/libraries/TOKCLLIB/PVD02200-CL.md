    VAR       ?PGMECX   ,STRING*11

    VAR       ?MSG      ,STRING*99(6)
    VAR       ?MSGX     ,STRING*99
    VAR       ?PGMID    ,STRING*8,VALUE-'PVD02200'

    OVRF      FILE-JYOKEN1,TOFILE-JYOKEN1.TOKFLIB
    CALL      PGM-SKY1601B.TOKELIB,PARA-(?WKSTN,?SOKO,?DSOKO)
    IF        @PGMEC    ^=   0    THEN

    OVRF      FILE-ZSOKMS1,TOFILE-ZSOKMS1.TOKFLIB

SORT01:
