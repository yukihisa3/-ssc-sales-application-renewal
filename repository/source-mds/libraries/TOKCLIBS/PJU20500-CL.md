    PARA ?JDATE   ,STRING*8,IN,VALUE-'00000000' /.受信日./

    PARA ?TORICD  ,STRING*8,IN,VALUE-'00000000' /.取引先CD./
    PARA ?JHFNM   ,STRING*8,IN,VALUE-'        ' /.受配信ファイル名./

    VAR  ?PGMECX   ,STRING*11                   /.ﾘﾀｰﾝｺｰﾄﾞ変換./
    VAR  ?PGMEM    ,STRING*99                   /.ﾘﾀｰﾝ名称./
    VAR  ?MSG      ,STRING*99(6)                /.ﾒｯｾｰｼﾞ退避ﾜｰｸ./

    VAR  ?PGMID    ,STRING*8,VALUE-'PJU20500'   /.PROGRAM-ID./
    VAR  ?STEP     ,STRING*8                    /.STEP-ID./
    VAR  ?SNDMSGR  ,STRING*20                   /.ﾒｯｾｰｼﾞ表示用./
    VAR  ?REBAK    ,INTEGER                     /.ﾊﾞｯｸｱｯﾌﾟﾏﾁ./
    VAR  ?PARAKEKA ,STRING*4,VALUE-'    '       /.ｴﾗｰ結果./
/.##ﾌｧｲﾙ変換ﾜｰｸ##./
    VAR  ?FILN     ,NAME                        /.ﾌｧｲﾙ名前型./
    VAR  ?LIBNM    ,NAME                        /.ﾗｲﾌﾞﾗﾘ名./
    VAR  ?JFILNM   ,NAME!MOD                    /.受信ﾌｧｲﾙ名./
    VAR  ?FILID    ,STRING*17                   /.ﾌｧｲﾙ名表示用./

    ACTMSGQ MAXRMSG-50

/.##プログラム開始メッセージ##./
    ?MSGX :=  '***   '  && ?PGMID  &&   ' START  ***'
    SNDMSG    ?MSGX,TO-XCTL

    SNDMSG MSG-'＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊',TO-XCTL
    SNDMSG MSG-'＊　受信データバックアップ開始　＊',TO-XCTL
    SNDMSG MSG-'＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊＊',TO-XCTL

    DEFLIBL TOKELIB/TOKFLIB/TOKJLIB/TOKELIBS/ONLBLIB

/.##ｴﾗｰ判断##./
FILENAME:

    ?STEP := 'FILENAME'
    ?MSGX := '***   '  && ?STEP   &&   '        ***'
    SNDMSG  ?MSGX,TO-XCTL

    ?FILN   := %NAME(?JHFNM)
    ?LIBNM  := %NAME(?LIBNMP)
    ?JFILNM := %NCAT(?FILN,?LIBNM)
    ?FILID  := %STRING(?JFILNM)
    ?MSGX   := '***受信ﾌｧｲﾙ名 = ' && ?FILID && ' ##'
    SNDMSG  ?MSGX,TO-XCTL

/.##資源獲得##./
PASSIGN:

    ?STEP :=   'PASSIGN '
    ?MSGX :=  '***   '  && ?STEP   &&   '        ***'
    SNDMSG    ?MSGX,TO-XCTL
    SNDMSG MSG-'**資源獲得',TO-XCTL
/.##ASSIGN FILE-BACK1200.ONLBLIB!@XCL

    IF   @PGMEC    ^=   0    THEN
         IF ?PGMEC = 4095 THEN
            IF ?REBAK < 5  THEN
               ?REBAK := ?REBAK + 1
               ?MSGX := '## 他JOBで資源使用中です BACK0128 ##'
               SNDMSG   ?MSGX,TO-XCTL
##./           /.### ﾊﾞｯｸｱｯﾌﾟ ﾏﾁｱﾜｾ ###./
/.##           CALL TWAIT30.XUCL
               GOTO   PASSIGN
            ELSE
               ?PARAKEKA  :=  'K532'
               GOTO   ABEND
            END
         ELSE
           ?PARAKEKA  :=  'K600'
           GOTO   ABEND
         END
    END
##./
/.##受信ファイルバックアップ##./
DATABACK:

    CALL SCVMSG.TOKCLIBO,PARA-('ｼﾞｭｼﾝF BAK 1200 START')
    ?STEP := 'DATABACK'
    ?MSGX := '***   '  && ?STEP   &&   '        ***'
    SNDMSG   ?MSGX,TO-XCTL

    OVRF FILE-CVCSG005,TOFILE-?JFILNM
    OVRF FILE-BKCVCS05,TOFILE-BACK1200.ONLBLIB

##./?PGMEC := @PGMEC
    ?PGMEM := @PGMEM
    IF  ?PGMEC ^= 0 THEN
        ?PARAKEKA  :=  'K600'

    END

/.##ﾌﾟﾛｸﾞﾗﾑ正常終了##./
RTN:

/.##RELEASE LIB-ONLBLIB!@XCL##./
    CALL SCVMSG.TOKCLIBO,PARA-('ｼﾞｭｼﾝF BAK 1200 END  ')
    ?SNDMSGR := ' ﾌｧｲﾙ名 = ' && ?JHFNM
    CALL SCVMSG.TOKCLIBO,PARA-(?SNDMSGR)
    ?MSGX := '***   ' && ?PGMID && ' END    ***'

/.##ﾌﾟﾛｸﾞﾗﾑ異常終了##./

    CALL SCVMSG.TOKCLIBO,PARA-('ｼﾞｭｼﾝF BAK 1200 ABEND')
    ?PGMECX := %STRING(?PGMEC)
    ?MSG(1) := '### ' && ?PGMID && ' ABEND ' && ' ###'
    ?MSG(2) := '### PGMEC = ' &&
               %SBSTR(?PGMECX,8,4) && '   ###'
    ?MSG(3) := '### LINE  = ' && %LAST(LINE) && '   ###'
    FOR ?I := 1 TO 3
      DO ?MSGX := ?MSG(?I)
         SNDMSG ?MSGX,TO-XCTL
    END

    SNDMSG ?PGMEM,TO-XCTL
    RETURN PGMEC-@PGMEC
