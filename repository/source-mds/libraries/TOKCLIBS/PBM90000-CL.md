/.##ﾗｲﾌﾞﾗﾘﾘｽﾄ登録##./

/.##ﾌﾟﾛｸﾞﾗﾑ名称ｾｯﾄ##./
    ?PGNM :=  '流通ＢＭＳ各種累積データ削除処理'

    IF  ?KBN = '2'   THEN
        GOTO SBM9000I
    ELSE
        GOTO SBM9080B
    END

    ?MSGX :=  '***   '  && ?STEP   &&   '        ***'
    SNDMSG    ?MSGX,TO-XCTL.@ORGPROF,JLOG-@YES

    OVRDSPF   FILE-DSPF,TOFILE-DSPF.XUCL,MEDLIB-TOKELIBO

                                         ?HIDUKE,
                                         ?JIKAN,
                                         ?TORCD,
                                         ?RCVDT)
    IF        @PGMEC    ^=   0    THEN
              IF  @PGMEC = 4010   THEN
                  ?MSGX := '##取消終了##'
                  SNDMSG ?MSGX,TO-XCTL.@ORGPROF,JLOG-@YES
                  RETURN
              ELSE
                  ?KEKA4 :=  '【削除条件入力】'
                  GOTO ABEND
              END
    ELSE
        /.  SNDMSG  ?DT    ,TO-XCTL
            SNDMSG  ?SYURUI,TO-XCTL
            SNDMSG  ?HIDUKE,TO-XCTL
            SNDMSG  ?JIKAN,TO-XCTL
            SNDMSG  ?TORCD,TO-XCTL
            SNDMSG  ?RCVDT,TO-XCTL   ./

            ?MSGX :=  '処理種別＝'  && '？？？'

                    ?MSGX :=  '処理種別＝'  && '定型'

                    ?MSGX :=  '削除対象＝'  && '発注'
            END
            IF      ?DT     = '02'  THEN
                    ?MSGX :=  '削除対象＝'  && '出荷'
            END
            IF      ?DT     = '03'  THEN
                    ?MSGX :=  '削除対象＝'  && '受領'
            END
            IF      ?DT     = '04'  THEN
                    ?MSGX :=  '削除対象＝'  && '受領訂正'
            END
            IF      ?DT     = '05'  THEN
                    ?MSGX :=  '削除対象＝'  && '返品'
            END

                    ?MSGX :=  '削除対象＝'  && '支払'
            END

                    ?MSGX :=  '削除対象＝'  && '請求'

                    ?MSGX :=  '指定種類＝'  && 'バッチNO'

          GOTO SBM9010B

    ?STEP :=   %LAST(LABEL)
