```jcl
/. ***********************************************************  ./
/. *     サカタのタネ　新基幹システム                        *  ./
/. *   SYSTEM-NAME :    入荷管理                             *  ./
/. *   JOB-ID      :    PNY00500                             *  ./
/. *   JOB-NAME    :    入荷実績リスト　                     *  ./
/. ***********************************************************  ./
    PGM
    VAR       ?PGMEC    ,INTEGER
    VAR       ?PGMECX   ,STRING*11
    VAR       ?PGMEM    ,STRING*99
    VAR       ?MSG      ,STRING*99(6)
    VAR       ?MSGX     ,STRING*99
    VAR       ?PGMID    ,STRING*8,VALUE-'PNY00500'
    VAR       ?STEP     ,STRING*8
    VAR       ?SOKO     ,STRING*2
    VAR       ?DSOKO    ,STRING*2
    VAR       ?WKSTN    ,STRING*8
    VAR       ?NWKSTN   ,NAME
    ?NWKSTN    :=       @ORGWS
    ?WKSTN    :=        %STRING(?NWKSTN)

    ?MSGX :=  '***   '  && ?PGMID  &&   ' START  ***'
    SNDMSG    ?MSGX,TO-XCTL

    DEFLIBL    TOKFLIB/TOKELIB

    CRTFILE   FILE-NYJISEKF.TOKFLIB,SIZE-100,ESIZE-10
    CHGFILE   FILE-NYJISEKF.TOKFLIB

/.  倉庫ＣＤ取得                                                ./
SKY1601B:
    ?STEP :=   'SKY1601B'
    ?MSGX :=  '***   '  && ?STEP   &&   '        ***'
    SNDMSG    ?MSGX,TO-XCTL

    OVRF      FILE-JYOKEN1,TOFILE-JYOKEN1.TOKFLIB
    CALL      PGM-SKY1601B.TOKELIB,PARA-(?WKSTN,?SOKO,?DSOKO)
    IF        @PGMEC    ^=   0    THEN
              GOTO ABEND END


/.  入荷実績リスト抽出                                          ./
SNY0050I:
    ?STEP :=   'SNY0050I'
    ?MSGX :=  '***   '  && ?STEP   &&   '        ***'
    SNDMSG    ?MSGX,TO-XCTL

    OVRDSPF   FILE-DSPF,TOFILE-DSPF.XUCL,MEDLIB-TOKELIB
    OVRF      FILE-NYKFILL2,TOFILE-NYKFILL2.TOKFLIB
    OVRF      FILE-HACHEDL1,TOFILE-HACHEDL1.TOKFLIB
    OVRF      FILE-HACMEIL1,TOFILE-HACMEIL1.TOKFLIB
    OVRF      FILE-JYOKEN1,TOFILE-JYOKEN1.TOKFLIB
    OVRF      FILE-ZSOKMS1,TOFILE-ZSOKMS1.TOKFLIB
    OVRF      FILE-NYJISEKF,TOFILE-NYJISEKF.TOKFLIB
    CALL      PGM-SNY0050I.TOKELIB,PARA-(?SOKO,?DSOKO)
    IF        @PGMEC     =   4050 THEN
              GOTO RTN   END
    IF        @PGMEC    ^=   0    THEN
              GOTO ABEND END

/.  ＳＯＲＴ                                                    ./
SORT01:
    ?STEP :=   'SORT'
    ?MSGX :=  '***   '  && ?STEP   &&   '        ***'
    SNDMSG    ?MSGX,TO-XCTL

    SORT       INFILE-NYJISEKF.TOKFLIB,INRL-150,INBF-6,
               OUTFILE-NYJISEKF.TOKFLIB,OUTBF-6,RCD-4,
               KEY-1!17!CA,KEY1-20!2!CA,KEY2-18!2!CA,KEY3-22!8!CA,
               RCDL-@DSP
    IF        @PGMEC    ^=   0    THEN
              GOTO ABEND END

/.  実績予定リスト出力                                          ./
SNY0060L:
    ?STEP :=   'ZNY0060L'
    ?MSGX :=  '***   '  && ?STEP   &&   '        ***'
    SNDMSG    ?MSGX,TO-XCTL

    OVRF      FILE-NYJISEKF,TOFILE-NYJISEKF.TOKFLIB
    OVRF      FILE-ZSHIMS1,TOFILE-ZSHIMS1.TOKFLIB
    OVRF      FILE-ZSOKMS1,TOFILE-ZSOKMS1.TOKFLIB
    OVRF      FILE-MEIMS1,TOFILE-MEIMS1.TOKFLIB
    CALL      PGM-SNY0060L.TOKELIB
    IF        @PGMEC    ^=   0    THEN
              GOTO ABEND END

RTN:

    ?MSGX :=  '***   '  && ?PGMID  &&   ' END    ***'
    SNDMSG    ?MSGX,TO-XCTL

    DLTFILE   NYJISEKF.TOKFLIB
    RETURN    PGMEC-@PGMEC

ABEND:

    ?PGMEC    :=    @PGMEC
    ?PGMEM    :=    @PGMEM
    ?PGMECX   :=    %STRING(?PGMEC)
    ?MSG(1)   :=   '### ' && ?PGMID && ' ABEND' &&   '    ###'
    ?MSG(2)   :=   '###' && ' PGMEC = ' &&
                    %SBSTR(?PGMECX,8,4) &&         '      ###'
    ?MSG(3)   :=   '###' && ' STEP = '  && ?STEP
                                                   && '   ###'


    FOR ?I    :=     1 TO 3
        DO     ?MSGX :=   ?MSG(?I)
               SNDMSG    ?MSGX,TO-XCTL
    END

    DLTFILE   NYJISEKF.TOKFLIB
    RETURN    PGMEC-@PGMEC
```
