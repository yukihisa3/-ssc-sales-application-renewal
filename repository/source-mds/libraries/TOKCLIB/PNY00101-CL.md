```jcl
/. ***********************************************************  ./
/. *     サカタのタネ　特販システム（本社システム）          *  ./
/. *   SYSTEM-NAME :    入庫管理システム                     *  ./
/. *   JOB-ID      :    SNY0010I                             *  ./
/. *   JOB-NAME    :    発注消込入力                         *  ./
/. ***********************************************************  ./
    PGM
/.##ﾜｰｸ定義##./
    VAR       ?PGMEC    ,INTEGER
    VAR       ?PGMECX   ,STRING*11
    VAR       ?PGMEM    ,STRING*99
    VAR       ?MSG      ,STRING*99(6)
    VAR       ?MSGX     ,STRING*99
    VAR       ?PGMID    ,STRING*8,VALUE-'PNY00100'
    VAR       ?STEP     ,STRING*8
    VAR       ?WKSTN    ,STRING*8,VALUE-'        '
    VAR       ?NWKSTN   ,NAME!MOD
/.##実行ﾜｰｸｽﾃｰｼｮﾝ取得
    ?NWKSTN    :=       @ORGWS
    ?WKSTN     :=       %STRING(?NWKSTN)
/.##ﾗｲﾌﾞﾗﾘﾘｽﾄ登録##./
    DEFLIBL    TOKELIB/TOKFLIB
/.##ﾌﾟﾛｸﾞﾗﾑ開始ﾒｯｾｰｼﾞ##./
    ?MSGX :=  '***   '  && ?PGMID  &&   ' START  ***'
    SNDMSG    ?MSGX,TO-XCTL
/.##実行倉庫ｺｰﾄﾞ表示##./
    ?MSGX :=  '##倉庫CD = ' &&  ?WKSTN && ' ##'
    SNDMSG    ?MSGX,TO-XCTL
/.##発注消込入力##./
SNY0010I:
    ?STEP :=   'SNY0010I'
    ?MSGX :=  '***   '  && ?STEP   &&   '        ***'
    SNDMSG    ?MSGX,TO-XCTL

/.##ﾌｧｲﾙ置き換え##./
    OVRDSPF   FILE-DSPF,TOFILE-DSPF.XUCL,MEDLIB-TOKELIB
    OVRF      FILE-HJYOKEN1,TOFILE-JYOKEN1.TOKFLIB
    OVRF      FILE-TOKMS1,TOFILE-TOKMS2.TOKFLIB
    OVRF      FILE-TENMS1,TOFILE-TENMS1.TOKFLIB
    OVRF      FILE-ZSHIMS1,TOFILE-ZSHIMS1.TOKFLIB
    OVRF      FILE-ZSOKMS1,TOFILE-ZSOKMS1.TOKFLIB
    OVRF      FILE-MEIMS1,TOFILE-MEIMS1.TOKFLIB
    OVRF      FILE-SHOTBL4,TOFILE-SHOTBL4.TOKFLIB
    OVRF      FILE-ZZAIMS1,TOFILE-ZZAIMS1.TOKFLIB
    OVRF      FILE-HACHEDL1,TOFILE-HACHEDL1.TOKFLIB
    OVRF      FILE-HACMEIL1,TOFILE-HACMEIL1.TOKFLIB
    OVRF      FILE-NYKFILL1,TOFILE-NYKFILL1.TOKFLIB
    CALL      PGM-SNY0010I.TOKELIB,PARA-(?WKSTN)
/.##ﾌﾟﾛｸﾞﾗﾑ終了ｺｰﾄﾞ判定##./
    IF        @PGMEC    ^=   0    THEN
              GOTO ABEND END

RTN:

    ?MSGX :=  '***   '  && ?PGMID  &&   ' END    ***'
    SNDMSG    ?MSGX,TO-XCTL

    RETURN    PGMEC-@PGMEC

ABEND:

    ?PGMEC    :=    @PGMEC
    ?PGMEM    :=    @PGMEM
    ?PGMECX   :=    %STRING(?PGMEC)
    ?MSG(1)   :=   '### ' && ?PGMID && ' ABEND' &&   '    ###'
    ?MSG(2)   :=   '###' && ' PGMEC = ' &&
                    %SBSTR(?PGMECX,8,4) &&         '      ###'
    ?MSG(3)   :=   '###' && ' STEP = '  && ?STEP
                                                   && '   ###'
    FOR ?I    :=     1 TO 3
        DO     ?MSGX :=   ?MSG(?I)
               SNDMSG    ?MSGX,TO-XCTL
    END

    RETURN    PGMEC-@PGMEC

```
