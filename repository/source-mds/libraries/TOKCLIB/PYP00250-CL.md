    VAR ?PGMEM    ,STRING*99

    VAR ?MSGX     ,STRING*99

    VAR ?OPR1     ,STRING*50                /.ﾒｯｾｰｼﾞ1    ./

    VAR ?ONLDTN   ,STRING*17

/.##ﾗｲﾌﾞﾗﾘﾘｽﾄ登録##./

    ?MSGX    :=  '## ﾜｰｸｽﾃｰｼｮﾝ名 = ' && ?WS
    SNDMSG MSG-?MSGX,TO-XCTL.@ORGPROF,JLOG-@YES

    OVRF      FILE-JYOKEN1,TOFILE-JYOKEN1.TOKFLIB
    CALL      PGM-SKY1601B.TOKELIB,PARA-(?WS,?SOKCD,?DSOKCD)

    END

    END

FILNMCHG:

    ?MSGX := '## 実行倉庫ｺｰﾄﾞ = ' && ?SOKCD && ' ##'

    FOR ?I    :=     1 TO 3
